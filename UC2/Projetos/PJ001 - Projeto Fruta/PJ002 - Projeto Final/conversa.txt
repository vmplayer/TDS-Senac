const bossOriginal = [
  "   /\\_/\\   ",
  "  ( o.o )  ",
  "   > ^ <   "
];

// Configurações do efeito
const LARGURA = 15;
const ALTURA = 5;
const QUADROS = 8; // Quantidade de etapas na animação

// 1. Cria uma matriz vazia (tela) para trabalhar os pixels textuais
let tela = Array(ALTURA).fill(null).map(() => Array(LARGURA).fill(" "));

// 2. Centraliza o Boss original dentro dessa tela
const offsetRow = Math.floor((ALTURA - bossOriginal.length) / 2);
bossOriginal.forEach((linha, r) => {
  for (let c = 0; c < linha.length; c++) {
    if (linha[c] !== " ") {
      tela[r + offsetRow][c] = linha[c];
    }
  }
});

let quadroAtual = 0;

const animacao = setInterval(() => {
  console.clear();

  // Exibe o estado atual da tela
  console.log(tela.map(linha => linha.join("")).join("\n"));

  // 3. Aplica a lógica de "derretimento" e estática para o próximo quadro
  for (let r = 0; r < ALTURA; r++) {
    for (let c = 0; c < LARGURA; c++) {
      let char = tela[r][c];

      // Ignora o que já sumiu
      if (char === " ") continue;

      // Probabilidade baseada no avanço da animação
      let chanceDistorcer = quadroAtual / QUADROS;

      if (Math.random() < chanceDistorcer) {
        if (char !== "|") {
          // Fase inicial: transforma partes do corpo do Boss em barras verticais
          tela[r][c] = "|";
        } else {
          // Fase intermediária: faz as barras "escorrerem" para baixo ou sumirem
          tela[r][c] = " ";
          if (r + 1 < ALTURA && Math.random() > 0.4) {
            tela[r + 1][c] = "|"; // Faz a linha vertical descer
          }
        }
      }
    }
  }

  quadroAtual++;

  // 4. Critério de parada
  if (quadroAtual > QUADROS) {
    clearInterval(animacao);
    console.clear();
    console.log("\n   O Boss foi apagado da existência...\n");
  }
}, 250); // Velocidade da transição (em milissegundos)
